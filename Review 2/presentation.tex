\begin{frame}{Technical Validation: Problem Formulation}
    \begin{columns}
        \begin{column}{0.6\textwidth}
            \textbf{Research Problem:}
            \textit{How to build a privacy-preserving, personalized, and explainable forecasting system that adapts to changing patient conditions in distributed healthcare environments?}
            
            \vspace{0.3cm}
            \textbf{Identified Gaps:}
            \begin{itemize}
                \item \textbf{Privacy vs. Collaboration:} Hospitals cannot share raw data.
                \item \textbf{Non-IID Data:} Patients in different hospitals have diverse demographics.
                \item \textbf{Concept Drift:} Patient health degrades/recovers over time (Concept Drift).
                \item \textbf{Transparency:} Clinicians need to know \textit{why} a patient is at risk.
            \end{itemize}
        \end{column}
        \begin{column}{0.4\textwidth}
            \centering

    \end{columns}
\end{frame}

% --- TECHNICAL DEPTH ---
\begin{frame}{Technical Depth: CUSUM & Attention Mechanics}
    \small
    \begin{columns}
        \begin{column}{0.5\textwidth}
            \textbf{1. Concept Drift Adaptation (CUSUM)}
            \[
            S_t = \max(0, S_{t-1} + (x_t - \mu - k))
            \]
            \begin{itemize}
                \item Monitors cumulative local prediction error.
                \item Triggers retraining of the \textbf{personality head} when $S_t > H$.
                \item Ensures the model remains accurate even as patient condition changes.
            \end{itemize}
        \end{column}
        \begin{column}{0.5\textwidth}
            \textbf{2. Explainable Attention}
            \[ \alpha_t = \frac{\exp(e_t)}{\sum \exp(e_k)} \]
            \begin{itemize}
                \item Assigns weights to specific timestamps in patient history.
                \item \textbf{Clinical Value:} Identifies that "Heart Rate surge 2 hours ago" was the primary risk factor.
            \end{itemize}
        \end{column}
    \end{columns}
\end{frame}
