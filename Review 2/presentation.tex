% --- INTRODUCTION AND BACKGROUND (2) ---
\begin{frame}{Introduction and Background}
    \textbf{Introduction to the Problem:}
    In modern Intensive Care Units (ICUs), continuous monitoring systems generate vast amounts of high-frequency multivariate time-series data (Heart Rate, Blood Pressure, SpO2). Detecting critical deterioration early saves lives.

    \vspace{0.3cm}
    \textbf{Motivation to Solve This Specific Problem:}
    \begin{itemize}
        \item \textbf{The Data Silo Problem:} High-accuracy models require massive datasets, but laws (e.g., \textbf{HIPAA} / \textbf{DPDPA 2023, Section 8: Obligations}) forbid combining data into a centralized server.
        \item \textbf{The Global Model Failure:} If multiple hospitals successfully engage in standard Federated Learning (FedAvg), the resulting "global model" fails because hospital demographics and local treatment protocols vary wildly.
        \item \textbf{The Black Box Problem:} Doctors cannot act on an AI alert if the model cannot explain the physiological basis for its warning.
    \end{itemize}
\end{frame}

% --- DATA PREPROCESSING (10) ---
\begin{frame}{Data Preprocessing \& Imputation Pipeline}
    \textbf{Challenge \& Why it looks like a mess:} 
    Real clinical records in PhysioNet contain massive sparsity and \texttt{NaN} gaps. High-frequency vitals (Heart Rate) are logged continuously by bedside monitors, while low-frequency lab tests (Bilirubin, Calcium) are only drawn manually by nurses every 12-24 hours.
    
    \vspace{0.2cm}
    \textbf{Engineered PyTorch Pipeline:}
    We developed a local Python preprocessing engine to dynamically ingest the 40,000+ \texttt{.psv} files and render clean PyTorch Tensors:
    \begin{enumerate}
        \item \textbf{Forward-Filling (Time-Series Carry):} Carries the last-known medical lab value forward in the sequence until a new blood measurement is taken.
        \item \textbf{Backward-Filling \& Zero-Padding:} Synthesizes initial missing conditions and replaces completely unmeasured lab variables with \texttt{0}s to prevent Neural Network failure.
        \item \textbf{Tensor Standardization:} Aggregates parameters via \texttt{StandardScaler} so highly varying units (e.g., Blood Pressure vs pH levels) do not skew the FPDAF gradient descent.
    \end{enumerate}
\end{frame}



% --- ALGORITHMIC NOVELTY (12) ---
\begin{frame}{Algorithmic Novelty \& Innovation Plan}
    \textbf{The Research Novelty of FPDAF:}
    We are not utilizing standard implementations of \textit{FedAvg} or \textit{CUSUM}. We are proposing a mathematical extension that fuses them into a single, responsive clinical algorithm.
    
    \vspace{0.3cm}
    \textbf{The Novel Plan:}
    \begin{itemize}
        \item \textbf{CUSUM as a Neural Trigger:} We mathematically repurpose \textit{CUSUM} from a simple statistical quality-control tool into an active Neural Network manipulator. When the formulation $S_t > H$ alerts, our novel code aggressively unfreezes the isolated Local Personalization Head, forcing immediate recalibration without destroying the global backbone.
        \item \textbf{Attention-Weighted Decentralization:} Standard \textit{FedAvg} treats all hospitals equally based on sample size. FPDAF will introduce a novel extension aggregating weights locally, prioritized by our \textit{Attention Mechanism}'s confidence, preventing outlier clinical anomalies from polarizing the central server.
    \end{itemize}
\end{frame}

% --- EXECUTION LOGIC DETAILS (16) ---
\begin{frame}{FPDAF Detailed Implementation Logic (What \& How)}
    \begin{itemize}
        \item \textbf{1. Decentralization (Global Syncing)}
        \begin{itemize}
            \item \textit{What:} Establish a privacy-preserving Baseline.
            \item \textit{How:} Code a central PyTorch Server script to securely aggregate only the \texttt{Global Temporal Base} gradient weights across 3-5 simulated hospital nodes, ensuring raw patient telemetry never leaves the client device.
        \end{itemize}
        \vspace{0.2cm}
        \item \textbf{2. Customization (Personalized Modeling)}
        \begin{itemize}
            \item \textit{What:} Adapt Sepsis predictions to specific hospital demographics.
            \item \textit{How:} Instantiate independent \texttt{Local Personalization Heads} (\textit{FedPer}) completely disconnected from the \textit{FedAvg} loop, allowing PyTorch client layers to heavily fit their local ICU distributions independently.
        \end{itemize}
        \vspace{0.2cm}
        \item \textbf{3. Statistical Math (Active Concept Drift)}
        \begin{itemize}
            \item \textit{What:} Enable active AI response to rapid patient deterioration.
            \item \textit{How:} We will sequentially track prediction residuals ($e_t$). If the $S_t > H$ CUSUM mathematical threshold breaches due to sudden physiological crashing, we dynamically unfreeze the personalization head for instant local patient recalibration!
        \end{itemize}
    \end{itemize}
\end{frame}
