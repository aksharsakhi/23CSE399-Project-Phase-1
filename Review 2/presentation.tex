% --- PROJECT PROPOSAL AND OBJECTIVES ---
\begin{frame}{Project Proposal and Objectives}
    \textbf{Overall Project Idea:}
    To design a completely decentralized, \textbf{Federated Personalized Drift-Aware Attention Framework (FPDAF)} tailored specifically for Multivariate Healthcare Time-Series (e.g., predicting ICU risks like Sepsis from patient vitals). 

    \vspace{0.3cm}
    \begin{block}{Objectives to Achieve}
        \begin{enumerate}
            \item \textbf{Privacy Preservation:} Train predictive models across multiple hospitals collaboratively without sharing raw patient data (Federated Learning).
            \item \textbf{Personalization:} Use Local Personalization Heads to manage patient heterogeneity (Non-IID distributions).
            \item \textbf{Drift Adaptation:} Implement CUSUM algorithms to detect when a patient's physiological state degrades and rapidly adapt the model locally.
            \item \textbf{Clinical Trust:} Provide transparency using Temporal Attention mechanisms so doctors know \textit{why} the model predicted a risk.
        \end{enumerate}
    \end{block}
\end{frame}

% --- INTRODUCTION AND BACKGROUND ---
\begin{frame}{Introduction and Background}
    \textbf{Introduction to the Problem:}
    In modern Intensive Care Units (ICUs), continuous monitoring systems generate vast amounts of high-frequency multivariate time-series data (Heart Rate, Blood Pressure, SpO2). Detecting critical deterioration early saves lives.

    \vspace{0.3cm}
    \textbf{Motivation to Solve This Specific Problem:}
    \begin{itemize}
        \item \textbf{The Data Silo Problem:} High-accuracy Deep Learning models require massive datasets, but rigid healthcare privacy laws (e.g., HIPAA) forbid combining data into a centralized server.
        \item \textbf{The Global Model Failure:} If multiple hospitals successfully engage in standard Federated Learning (FedAvg), the resulting "global model" fails because hospital demographics and local treatment protocols vary wildly.
        \item \textbf{The Black Box Problem:} Doctors cannot act on an AI alert if the model cannot explain the physiological basis for its warning.
    \end{itemize}
\end{frame}

% --- DATASET USED ---
\begin{frame}{Dataset: PhysioNet Challenge 2019 (Sepsis Prediction)}
    \begin{columns}
        \begin{column}{0.65\textwidth}
            \textbf{Dataset Overview:}
            \begin{itemize}
                \item \textbf{Source:} PhysioNet (Public Healthcare Dataset).
                \item \textbf{Data Type:} Real-world multivariate ICU patient monitoring time-series.
                \item \textbf{Structure:} ~40,000 patient files (\texttt{.psv}), with each row representing one hourly observation.
                \item \textbf{Key Features:} Heart Rate, BP, SpO2, Temperature, ICULOS.
                \item \textbf{Target:} \texttt{SepsisLabel} (predicting deterioration).
            \end{itemize}
        \end{column}
        \begin{column}{0.35\textwidth}
            \begin{block}{Why PhysioNet?}
                \small
                \begin{itemize}
                    \item \textbf{Heterogeneity:} \texttt{setA} \& \texttt{setB} come from different hospitals (perfect for FL).
                    \item \textbf{Time-Series:} Sequences support LSTM \& Drift tracking.
                    \item \textbf{Clinical:} Interpretable vitals support our XAI mechanism.
                \end{itemize}
            \end{block}
        \end{column}
    \end{columns}
\end{frame}

\begin{frame}{Architecture Diagram and Explanation of Design}
    \begin{columns}
        \begin{column}{0.4\textwidth}
            \textbf{Why this design?}
            \small
            \begin{itemize}
                \item \textbf{Privacy:} Only encrypted gradients travel upward.
                \item \textbf{Local LSTM Base:} Captures generalized physiological rules.
                \item \textbf{Attention Layer:} Provides XAI (feature importance) right before the local prediction.
                \item \textbf{Personalization Head:} NEVER shared with the server. Tailors risk predictions to the specific local patient cohort.
