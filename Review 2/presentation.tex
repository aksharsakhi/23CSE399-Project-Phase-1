\begin{frame}{Targeted Problem Statement: Smart Grids}
    \begin{columns}
        \begin{column}{0.6\textwidth}
            \begin{block}{The Challenge}
                Accurate Short-Term Load Forecasting (STLF) is hindered by:
                \begin{itemize}
                    \item \textbf{Privacy:} Household energy data is legally protected.
                    \item \textbf{Heterogeneity:} Non-IID profiles (High-usage vs. Solar-homes).
                    \item \textbf{Drift:} Behavioral shifts due to seasons or appliance upgrades.
                \end{itemize}
            \end{block}
            \centering \small \textit{Objective: A framework that learns globally but adapts locally.}
        \end{column}

\begin{frame}{Domain Context: Pecan Street Dataset}
    \begin{itemize}
        \item \textbf{Nature:} The world's largest research-grade smart grid dataset.
        \item \textbf{Features:} Multivariate time-series (Voltage, Power, Solar Gen, Temp).
        \item \textbf{Granularity:} 15-minute intervals covering 100+ residential homes.
        \item \textbf{Why it fits?} Perfect for simulating \textbf{client heterogeneity} (EV owners, student dorms, net-zero houses) in a Federated setting.
    \end{itemize}
    \begin{block}{The Non-IID Problem}
        Local distributions differ significantly — standard global averaging (FedAvg) leads to high variance and unstable local convergence.
    \end{block}

% --- ATTENTION MATHEMATICS ---
\begin{frame}{Internal Mechanics: Explainable Attention}
    \small
    \textbf{Problem:} Black-box LSTMs don't explain why a peak was predicted.
    \textbf{Our Solution:} Attention weights that assign importance to historical lags.

    \begin{block}{Attention Score Calculation}
        \[ e_t = \text{score}(h_t, h_T) \]
        \[ \alpha_t = \frac{\exp(e_t)}{\sum_{k=1}^{T} \exp(e_k)} \]
    \end{block}

    \begin{itemize}
        \item \textbf{High $\alpha_t$:} Indicates the model is relying on that specific hour (e.g., "Last Monday at 6 PM").
        \item \textbf{Trust:} Grid operators can verify if the prediction aligns with known trends (e.g., Cooking peaks, AC usage).
    \end{itemize}
\end{frame}
